=== VERIFICAÇÃO DE INTEGRIDADE ===

mysql> SELECT 
    'Verificação de integridade' as status,
    COUNT(*) as total_leituras,
    COUNT(DISTINCT id_sensor) as sensores_unicos,
    MIN(timestamp_datetime) as primeira_leitura,
    MAX(timestamp_datetime) as ultima_leitura
FROM leituras_sensores;

=== VERIFICAÇÃO DE FOREIGN KEYS ===

mysql> SELECT 
    'Verificação FK' as status,
    COUNT(*) as total_leituras,
    COUNT(CASE WHEN s.id_sensor IS NOT NULL THEN 1 END) as leituras_com_sensor_valido,
    ROUND(COUNT(CASE WHEN s.id_sensor IS NOT NULL THEN 1 END) * 100.0 / COUNT(*), 2) as percentual_validas
FROM leituras_sensores l
LEFT JOIN sensores s ON l.id_sensor = s.id_sensor;

+------------------+------------------+------------------+------------------+
| status           | total_leituras   | leituras_com_sensor_valido | percentual_validas |
+------------------+------------------+------------------+------------------+
| Verificação FK   | 1728000          | 1728000          | 100.00           |

=== VERIFICAÇÃO DE ALERTAS ===

mysql> SELECT 
    'Verificação Alertas' as status,
    COUNT(*) as total_alertas,
    COUNT(CASE WHEN a.id_dispositivo IS NOT NULL THEN 1 END) as alertas_com_dispositivo,
    COUNT(CASE WHEN a.id_sensor IS NOT NULL THEN 1 END) as alertas_com_sensor,
    ROUND(COUNT(CASE WHEN a.id_dispositivo IS NOT NULL THEN 1 END) * 100.0 / COUNT(*), 2) as percentual_dispositivo
FROM alertas a;
